Interpretação do IMC (faixas de referência)¶

  • Abaixo do peso: IMC < 18.5
  • Peso normal: 18.5 <= IMC < 24.9
  • Sobrepeso: 25 <= IMC < 29.9
  • Obesidade grau 1: 30 <= IMC < 34.9
  • Obesidade grau 2: 35 <= IMC < 39.9
  • Obesidade grau 3: IMC >= 40

# -----------------------------------------------------------------------------
# 0. Preparação dos Dados
# -----------------------------------------------------------------------------

# Cria uma cópia do DataFrame original (supondo que 'dados' já contenha a coluna 'Faixa_IMC'
# e 'Tempo_sono'; a coluna 'faixa_sono' será criada on the fly para agrupar Tempo_sono)
dados_binned = dados.copy()

# Se necessário, converte a coluna Faixa_IMC para categorical com a ordem desejada
desired_imc_order = [
    "Abaixo do peso",
    "Peso normal",
    "Sobrepeso",
    "Obesidade grau 1",
    "Obesidade grau 2",
    "Obesidade grau 3"
]
dados_binned['Faixa_IMC'] = pd.Categorical(
    dados_binned['Faixa_IMC'],
    categories=desired_imc_order,
    ordered=True
)

# -----------------------------------------------------------------------------
# 1. Agrupamento dos Dados usando as Regras "On The Fly" para Tempo_sono
# -----------------------------------------------------------------------------

# Aplica o binning on the fly para Tempo_sono conforme as regras:
# [0, 6)   -> Baixo; [6, 8) -> Médio; [8, 12] ou maiores -> Alto.
# (Aqui usamos pd.cut para criar a categorização temporária)
faixa_sono = pd.cut(
    dados_binned['Tempo_sono'],
    bins=[0, 6, 8, 12],
    labels=['Baixo', 'Médio', 'Alto'],
    right=False
)

# Cria uma pivot table agrupando por Faixa_IMC e pela categorização de Tempo_sono
pivot = dados_binned.groupby('Faixa_IMC')['Tempo_sono'].apply(
    lambda x: pd.cut(
        x, bins=[0, 6, 8, 12], labels=['Baixo', 'Médio', 'Alto'], right=False
    ).value_counts()
).unstack(fill_value=0)

# Reindexa o pivot para garantir a ordem desejada, tanto para linhas quanto para colunas
pivot = pivot.reindex(desired_imc_order, fill_value=0)
pivot = pivot[['Baixo', 'Médio', 'Alto']]

print(pivot)

# Cria uma cópia do DataFrame original
df_tela = dados.copy()

# Define a ordem desejada para Faixa_IMC (usando somente as faixas relacionadas a sobrepeso e obesidade)
desired_imc_order = [
    "Sobrepeso",
    "Obesidade grau 1",
    "Obesidade grau 2",
    "Obesidade grau 3"
]

# Força a ordenação da coluna Faixa_IMC para garantir a ordem desejada
df_tela['Faixa_IMC'] = pd.Categorical(
    df_tela['Faixa_IMC'],
    categories=desired_imc_order,
    ordered=True
)

# -----------------------------------------------------------------------------
# 1. Agrupamento dos Dados usando o Binning on the Fly para Tempo_tela
# -----------------------------------------------------------------------------

# Agrupa por Faixa_IMC e aplica o binning para a coluna Tempo_tela conforme as regras:
# Intervalo [0, 6) -> 'Baixo'
# Intervalo [6, 12) -> 'Médio'
# Intervalo [12, 18) -> 'Alto'
pivot_tela = df_tela.groupby('Faixa_IMC')['Tempo_tela'].apply(
    lambda x: pd.cut(x, bins=[0, 6, 12, 18], labels=['Baixo','Médio','Alto'], right=False).value_counts()
).unstack(fill_value=0)

# Reindexa o pivot para garantir a ordem desejada para Faixa_IMC
pivot_tela = pivot_tela.reindex(desired_imc_order, fill_value=0)

# Garantir a ordem das colunas de tempo de tela
pivot_tela = pivot_tela[['Baixo','Médio','Alto']]

print(pivot_tela)

# -----------------------------------------------------------------------------
# 2. Criação do Gráfico de Barras Agrupadas (Grouped Bar Chart)
# -----------------------------------------------------------------------------

fig = go.Figure()

# Adiciona um traço para cada categoria de Tempo_tela
for tempo_cat in ['Baixo', 'Médio', 'Alto']:
    fig.add_trace(go.Bar(
        x=pivot_tela.index.astype(str),  # Faixa_IMC no eixo x
        y=pivot_tela[tempo_cat],
        name=tempo_cat
    ))

# Configura o layout do gráfico
fig.update_layout(
    barmode='group',  # barras lado a lado para cada faixa de IMC
    title="Relação entre Faixa_IMC e Tempo de Tela",
    xaxis_title="Faixa de IMC",
    yaxis_title="Contagem",
    legend_title="Tempo de Tela"
)

# --- Preparação dos Dados ---
# Supondo que 'dados' seja seu DataFrame original com as colunas 'Tempo_sono' e 'Tempo_tela'.

# Aplica o binning "on the fly" para o Tempo de Sono:
sleep_bins = pd.cut(
    dados['Tempo_sono'],
    bins=[0, 6, 8, 12],
    labels=['Baixo', 'Médio', 'Alto'],
    right=False
)

# Aplica o binning "on the fly" para o Tempo de Tela:
screen_bins = pd.cut(
    dados['Tempo_tela'],
    bins=[0, 6, 12, 18],
    labels=['Baixo', 'Médio', 'Alto'],
    right=False
)

# Cria colunas temporárias com os bins (sem modificar o DataFrame original se preferir, mas para facilitar o agrupamento)
dados_temp = dados.copy()
dados_temp['faixa_sono'] = sleep_bins
dados_temp['faixa_tela'] = screen_bins

# --- Agrupamento dos Dados ---
# Agrupa por "faixa_sono" e para cada grupo conta quantos registros possuem cada categoria de "faixa_tela"
pivot_table = dados_temp.groupby('faixa_sono')['faixa_tela'].value_counts().unstack(fill_value=0)

# Define a ordem desejada (para assegurar consistência na exibição)
sleep_order = ['Baixo', 'Médio', 'Alto']
screen_order = ['Baixo', 'Médio', 'Alto']

# Reindexa para garantir a ordem desejada:
pivot_table = pivot_table.reindex(index=sleep_order, fill_value=0)
pivot_table = pivot_table[screen_order]

print(pivot_table)

# --- Criação do Gráfico de Barras Agrupadas ---
fig = go.Figure()

# Para cada categoria de Tempo de Tela, adiciona uma trace de barra (barras agrupadas)
for screen_cat in screen_order:
    fig.add_trace(go.Bar(
        x=pivot_table.index.astype(str),  # As faixas de Tempo de Sono ficam no eixo x
        y=pivot_table[screen_cat],
        name=screen_cat  # Cada barra recebe a legenda conforme a categoria de Tempo de Tela
    ))

fig.update_layout(
    barmode='group',  # Barras lado a lado para cada faixa de Tempo de Sono
    title="Distribuição de Tempo de Tela por Faixa de Tempo de Sono",
    xaxis_title="Faixa de Tempo de Sono",
    yaxis_title="Contagem",
    legend_title="Tempo de Tela"
)

fig.show()

Boxplot:¶

Utilizamos o Plotly Express (px.box) para criar um boxplot em que o eixo x representa as categorias de “Alimentação” e o y mostra os valores de IMC. Assim, é possível comparar visualmente a dispersão do IMC em cada grupo.¶

Índice de Saudabilidade:¶

Como a variável “Alimentação” é qualitativa, criamos um mapeamento para convertê-la em uma variável numérica. Quanto melhor a alimentação, maior o índice atribuído (por exemplo, “Muito saudável” → 4).¶

Gráfico de Dispersão:¶

Com o px.scatter relacionamos a variável contínua “Indice_saudabilidade” com o IMC. Assim, é possível verificar se existe alguma tendência, por exemplo, se um índice mais alto (alimentação melhor) está associado a um IMC menor ou maior.¶

# 1. Preservação do DataFrame original e criação de cópia para análise
# =============================================================================
dados_binned = dados.copy()

# Calcula o IMC (Peso em kg / (Altura em m)²)
dados_binned['IMC'] = dados_binned['Peso'] / ((dados_binned['Altura'] / 100) ** 2)

# =============================================================================
# 2. Boxplot: Distribuição do IMC por Categoria de Alimentação
# =============================================================================
fig_box = px.box(dados_binned, 
                 x="Alimentação", 
                 y="IMC",
                 title="Distribuição do IMC por Categoria de Alimentação",
                 labels={"Alimentação": "Categoria de Alimentação", "IMC": "IMC"})
fig_box.update_traces(marker=dict(color="dodgerblue"))

# =============================================================================
# 3. Cálculo dos Quartis Globais de IMC
# =============================================================================
q1 = dados_binned['IMC'].quantile(0.25)
median = dados_binned['IMC'].quantile(0.5)
q3 = dados_binned['IMC'].quantile(0.75)

# =============================================================================
# 4. Adição de Faixa e Linha para os Quartis
# =============================================================================
# Faixa (retângulo) de Q1 a Q3
fig_box.add_shape(
    type="rect",
    xref="paper", yref="y",
    x0=0, x1=1,
    y0=q1, y1=q3,
    fillcolor="red",
    opacity=0.2,
    layer="below"
)

# Linha horizontal na mediana
fig_box.add_shape(
    type="line",
    xref="paper", yref="y",
    x0=0, x1=1,
    y0=median, y1=median,
    line=dict(color="red", width=2)
)

# =============================================================================
# 5. Adição de Anotações com Ajustes de Posição
# =============================================================================
# Para as anotações, vamos usar um deslocamento horizontal maior (x=1.03) e ajustar o yshift

# Anotação para Q1 (um pouco acima ou abaixo para destaque)
fig_box.add_annotation(
    x=1.03, y=q1,
    xref="paper", yref="y",
    text=f"Q1: {q1:.2f}",
    showarrow=False,
    font=dict(color="red", size=12),
    yshift=-5  # deslocamento vertical para mover um pouco para baixo
)

# Anotação para Mediana (deslocamento vertical maior para não ficar sobre a linha)
fig_box.add_annotation(
    x=1.03, y=median,
    xref="paper", yref="y",
    text=f"Mediana: {median:.2f}",
    showarrow=False,
    font=dict(color="red", size=12),
    yshift=15  # deslocamento vertical para mover um pouco para cima
)

# Anotação para Q3
fig_box.add_annotation(
    x=1.03, y=q3,
    xref="paper", yref="y",
    text=f"Q3: {q3:.2f}",
    showarrow=False,
    font=dict(color="red", size=12),
    yshift=5  # deslocamento vertical para mover um pouco para cima
)

# Anotação extra para indicar que a linha vermelha é a mediana
fig_box.add_annotation(
    x=0.5, y=median,
    xref="paper", yref="y",

    font=dict(color="red", size=12),

# Supondo que já tenha sido criado anteriormente o "Índice de Saudabilidade" a partir da variável "Alimentação"
# Se não, você pode criar conforme:
mapping = {
    "Nada saudável": 0,
    "Pouco saudável": 1,
    "Saudável": 2,
    "Razoavelmente saudável": 3,
    "Muito saudável": 4
}
dados_binned['Indice_saudabilidade'] = dados_binned['Alimentação'].map(mapping)

# Cria o gráfico de dispersão relacionando o Índice de Saudabilidade com o IMC.
# Adiciona, opcionalmente, uma linha de tendência (regressão linear) para ver possíveis tendências.
fig_scatter = px.scatter(
    dados_binned,
    x="Indice_saudabilidade",
    y="IMC",
    title="Dispersão: Índice de Saudabilidade vs. IMC",
    labels={"Indice_saudabilidade": "Índice de Saudabilidade", "IMC": "IMC"},
    trendline="ols",          # Adiciona uma linha de regressão linear (opcional)
    hover_data=["Alimentação"]  # Exibe a categoria de alimentação ao pairar
)

# Ajusta o tamanho e a cor dos marcadores
fig_scatter.update_traces(marker=dict(size=8, color="firebrick"))
